def <=>(another)
    if(another.caloric_value > caloric_value)
        return -1
    else
        if(another.caloric_value < caloric_value)
            return 1
        end
        if (another.proteins > proteins)
            return -1
        else
            if(another.proteins < proteins)
                return 1
            end
            if(another.glucids > glucids)
                return -1
            else
                if(another.glucids < glucids)
                    return 1
                end
                if(another.lipids > lipids)
                    return -1
                else 
                    if(another.lipids < lipids)
                        return 1
                    else return 0
                    end
                end
            end
        end
    end
end

def caloric_value()
   return (@proteins * 4 + @lipids * 9 + @glucids * 4)
end
